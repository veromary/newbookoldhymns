
\beginhymn Sweet Sacrament divine.

\moveleft4mm\hbox{

\vtop{
\hsize54mm

{\leftskip1em
\textinitial{S}{weet} Sacrament divine,\\}
\beginlyrics
\qquad Hid in Thy earthly home,\\
Lo! round Thy lowly shrine,\\
\quad With suppliant hearts we come.\\
Jesus, to Thee our voice we raise,\\
In songs of love and heartfelt praise.\\
\quad Sweet Sacrament divine.


\verse 2. Sweet Sacrament of peace,\\
\quad Dear home of every heart,\\
Where restless yearnings cease,\\
\quad And sorrows all depart.\\
There in Thine ear all trustfully\\
We tell our tale of misery,\\
\quad Sweet Sacrament of peace.

\endlyrics
}
\hskip5mm
\vtop{
\hsize55mm
\beginlyrics

\verse3. Sweet Sacrament of rest,\\
\quad Ark from the ocean's roar,\\
Within Thy shelter blest\\
\quad Soon may we reach the shore.\\
Save us, for still the tempest raves;\\
Save, lest we sink beneath the waves,\\
\quad Sweet Sacrament of rest.

\verse4. Sweet Sacrament divine,\\
\quad Earth's light and jubilee,\\
In Thy far depths doth shine\\
\quad Thy Godhead's majesty.\\
Sweet light, so shine on us, we pray,\\
That earthly joys may fade away,\\
\quad Sweet Sacrament divine.

\endlyrics
}
}
